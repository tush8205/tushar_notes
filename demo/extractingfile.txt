BEGIN

LOOP
	

	lv_ubp_up := '';


	lv_ubp_up  :=  i.emp_id || ','||i.cust_id||','||i.cust_first_name||','||i.cust_last_name|| ','||i.cust_phoneNum||','||i.cust_age||','||i.cust_current_status ;

END loop;

END;
